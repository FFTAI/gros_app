import gettext



[文档]class i18n:
    ''' Internationalization class'''

    def __init__(self, lang: str = "zh-CN"):
        ''' lang could be zh-CN or en '''
        self.lang = lang
        self.lt = gettext.translation("base", "locales", languages=[lang], codeset='utf-8')
        self.lt.install()


[文档]    def txt(self, msg: str) -> str:
        ''' translate '''
        return self.lt.gettext(msg)
