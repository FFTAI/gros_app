import os
from configparser import ConfigParser

from base.singleton import Singleton

cwd = os.getcwd()
config_path = os.path.join(cwd, "config/robot.conf").replace("/doc", "")



[文档]class Config(Singleton):
    """配置文件处理类"""

    parser = ConfigParser()
    """ConfigParser implementing interpolation"""

    @property
    def robot_type(self) -> str:
        """获取机器类型

        Returns:
            str: 机器类型 car 、dog 、human
        """
        self.parser.read(config_path)
        robot_type = self.parser.get('Robot', 'type')
        return robot_type

    @property
    def robot_host_port(self) -> tuple[str, int]:
        """获取机器udp控制端的host和port

        Returns:
            tuple[str, int]: 机器端口号和port
        """
        self.parser.read(config_path)
        host = self.parser.get('Robot', 'udp_host')
        port = self.parser.get('Robot', 'udp_port')
        return host, int(port)
