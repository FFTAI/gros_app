[文档]    @staticmethod
    def reset():
        """重置wifi信息

        默认配置在config/wifi.conf.default文件

        ``重置之后将会重启电脑``
        """
        with open("config/wifi.conf.default", "r") as default_wifi_file, open("config/wifi.conf",
                                                                              "w") as target_wifi_file:
            content = default_wifi_file.read()
            target_wifi_file.write(content)



[文档]    @staticmethod
    def set_wifi(name: str, password: str):
        """
        设置wifi信息

        Args:
            name:
                str：wifi名称
            password:
                str: wifi密码

        ``设置成功之后将会重启电脑``
        """
        with open("config/wifi.conf", "w") as target_wifi_file:
            content = f'wifi_name="{name}"\nwifi_password="{password}"'
            target_wifi_file.write(content)
