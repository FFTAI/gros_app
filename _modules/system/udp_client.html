import json
import socket
from typing import Any



[文档]class UDPClient:
    """udp客户端"""

    def __init__(self, ip: str, port: int):
        self.ip = ip
        self.port = port
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self.server_address = (ip, port)

    def __enter__(self):
        return self

    def __exit__(self, exc_type, exc_value, traceback):
        self.sock.close()


[文档]    def send_message(self, message: Any):
        """
        发送消息：直接发送，传入的是什么类型就是什么类型

        Args:
            message:
                Any: 消息内容: 支持任意类型的消息
        """
        self.sock.sendto(bytes(message, "utf-8"), self.server_address)
        print(f"udp send_message: {message}")



[文档]    def send_json(self, message: Any):
        """
        发送json消息

        Args:
            message:
                Any: 消息内容: 支持任意类型的消息
        """
        self.send_message(json.dumps(message))



[文档]    def receive_message(self) -> object:
        """
        接收udp消息

        消息缓冲 102400KB 满足大文本

        Returns:
            object: 消息体
        """
        data, addr = self.sock.recvfrom(102400)
        print("udp receive_message: {}".format(data.decode("utf-8")))
        result = data.decode("utf-8")
        return result



[文档]    def send_json_receive(self, message: object):
        try:
            self.send_json(message)
            answer_json = self.receive_message()
            return answer_json
        except Exception as e:
            # 处理接收消息失败的情况
            print("Failed to receive message:", str(e))
            return None
