from functools import cached_property

from pydantic import BaseModel

from system.config import Config
from system.udp_client import UDPClient



[文档]class RobotUDP(BaseModel):
    """ 第一层 第二层 """

    @cached_property
    def udp(self) -> UDPClient:
        return self._connect()

    def _connect(self) -> UDPClient:
        """ 子类可以复写此函数，用不同方式返回UDPClient对象 
        但通常应该不需要
        """
        cfg = Config()
        self._udp = UDPClient(*cfg.robot_host_port)
        return self._udp
