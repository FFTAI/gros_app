    def _task_work_mode_udp(self, mod: int):
        """

        self.udp.send_json({"task": {"work_mode": mod}})

    def _flw_task_remote_whl_4_22_mc_control_udp(self, direction: float, velocity: float):
        """
        4轮模式下小车控制

        最底层发送udp消息驱动控制端: 驱动电机移动

        ``注意: 该操作对于电机的指定是不连续的，如果只发送了一次请求，那么对应的电机在没有收到后续请求的时候的只会持续执行最后一次请求``

        Args:
            direction(float): 角度
            velocity(float): 速度
        """
        self.udp.send_json({"flw": {"task_remote_whl_4_22_mc_control": {"target_direction": direction,
                                                                        "target_velocity": velocity}}})

[文档]    def speed(self, mod: CarMod) -> float:
        """
        获取小车当前速度

        Args:
            mod(CarMod): 模式

        Returns:
            float: 小车实时速度

        Raises:
            GROSError(RobotError): 模式错误
        """
        speed_ret_json = {}
        if mod == CarMod.MOD_4_WHEEL:
            speed_ret_json = self._flw_task_remote_whl_4_22_mc_speed_udp()
        elif mod == CarMod.MOD_3_WHEEL:
            speed_ret_json = self._flw_task_remote_whl_3_12_mc_speed_udp()
        elif mod == CarMod.MOD_2_WHEEL:
            speed_ret_json = self._flw_task_remote_whl_2_02_mc_speed_udp()
        else:
            raise GROSError(RobotError.MOD_NOT_ALLOWED)

        return speed_ret_json['flw'][mod.value["ctrl_val"]]['measured_body_posture'][0]



[文档]    def move(self, mod: CarMod, angle: float, speed: float):
        """移动

        Args:
            mod(CarMod): 模式
            angle(float): 角度
            speed(float): 速度

        Raises:
            GROSError(RobotError): 值错误

        """
        if not -90 <= angle <= 90:
            raise GROSError(RobotError.ANGLE_NOT_ALLOWED)
        direction = 1 - (angle / 90)

        if not -1 <= speed <= 1:
            raise GROSError(RobotError.ANGLE_NOT_ALLOWED)

        if mod == CarMod.MOD_4_WHEEL:
            self._flw_task_remote_whl_4_22_mc_control_udp(direction, speed)
        elif mod == CarMod.MOD_3_WHEEL:
            self._flw_task_remote_whl_3_12_mc_control_udp(direction, speed)
        elif mod == CarMod.MOD_2_WHEEL:
            self._flw_task_remote_whl_2_02_mc_control_udp(direction, speed)
        else:
            raise GROSError(RobotError.MOD_NOT_ALLOWED)
