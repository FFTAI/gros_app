    def _start_udp(self):
        """启动"""
        self.udp.send_json({"function": "SonnieCommand", "data": {"start": {}}})

    def _zero_udp(self):
        """回零"""
        self.udp.send_json({"function": "SonnieCommand", "data": {"zero": {}}})

    def _stop_udp(self):
        """回零"""
        self.udp.send_json({"function": "SonnieCommand", "data": {"stop": {}}})

    def _stand_udp(self):
        """stand模式下静止状态"""
        self.udp.send_json({"function": "SonnieCommand", "data": {"stand": {}}})

    def _stand_pos_cmd_head_udp(self, a: float, b: float, c: float):
        """stand模式下发送头部指令

        Args:
            a(float): Roll
            b(float): Pitch
            c(float): Yaw
        """
        self.udp.send_json(
            {"function": "SonnieCommand", "data": {"stand": {"pos_cmd": {"head": {"a": a, "b": b, "c": c}}}}})

    def _walk_udp(self, vx: float, vy: float, vz: float, va: float, vb: float, vc: float, ):
        """行走

        Args:
            vx(float): 向前向后速度值
            vy(float):
            vz(float):
            va(float):
            vb(float):
            vc(float): 左右旋转速度值
        """
        self.udp.send_json({"function": "SonnieCommand", "data": {
            "walk": {"vel_cmd": {"vx": vx, "vy": vy, "vz": vz, "va": va, "vb": vb, "vc": vc, }}}})

    def _get_states(self, frequence: int):
        """获取机器人状态

        Args:
            frequence(int): 频率
        """
        self.udp.send_json({"function": "SonnieGetStates", "data": {"frequence": frequence}})

    def _get_system_states(self):
        """获取系统状态"""
        self.udp.send_json({"function": "SonnieGetSystemStates", "data": {}})

    def _get_states_limit(self):
        """获取系统就位状态"""
        self.udp.send_json({"function": "SonnieGetStatesLimit", "data": {}})

[文档]    def stand_up(self, data):
        ''' 站立 '''
        # 调用之前校验入参是否合理
        stand_validate: Validator = Validator(stand_schema)
        print("校验stand参数")
        if stand_validate.validate(data):
            print("Data is valid!")
        else:
            print("reqeset_param:{} is invalid, error message is:{}".format(data, stand_validate.errors))
            return fgr_model.response_fail(stand_validate.errors)
        self.udp.send_json({
            "function": "SonnieCommand",
            "data": {"stand": data}
        })
        return fgr_model.response_ok(None)
