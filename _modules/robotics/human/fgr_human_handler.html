async def _walk(host, port, ws, message):
    data = {
        "vel_cmd": {
            "vx": message['walk_vx'] if 'walk_vx' in message else None,
            "vy": message['walk_vy'] if 'walk_vy' in message else None,
            "vz": message['walk_vz'] if 'walk_vz' in message else 0,
            "va": message['walk_va'] if 'walk_va' in message else 0,
            "vb": message['walk_vb'] if 'walk_vb' in message else 0,
            "vc": message['walk_vc'] if 'walk_vc' in message else None
        }
    }
    # 校验
    result: Result = dh.robot.walk(data)
    # 发送通知给客户端
    await ws.send_json(result)
