[文档]    def gen(self):
        while True:
            if not self.cap.grab():
                break
            ret1, frame = self.cap.retrieve()
            ret1, jpeg = cv2.imencode('.jpg', frame)
            jpg_frame = jpeg.tobytes()
            yield (b'--frame\r\n'
                   b'Content-Type: image/jpeg\r\n\r\n' + jpg_frame + b'\r\n')
