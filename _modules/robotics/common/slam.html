import asyncio
import platform
from os import path

from rplidar import RPLidar

# todo 1: slam 放入 robotics目录？因为 slam 是机器人的一部分

os_name = platform.system()

datas = dict()



[文档]async def run():
    dev_path = '/dev/ttyUSB0'
    if os_name == 'Linux':
        dev_path = '/dev/ttyUSB0'
    if os_name == 'Windows OS':
        dev_path = 'com5'
    if os_name == 'Darwin':
        dev_path = '/dev/tty.usbserial-0001'

    if path.exists(dev_path):
        lidar = RPLidar(dev_path)
        # 必须要先停一下，否则重启可能启动不成功
        lidar.stop()
        print("slam init success: OS: {} ".format(os_name))
        try:
            for val in lidar.iter_scans():
                for (_, angle, distance) in val:
                    if distance > 0:
                        await asyncio.sleep(0.001)
                        datas[str(int(angle))] = distance
        except:
            lidar.stop()
            lidar.stop_motor()
            lidar.disconnect()

    else:
        print(f"slam init fail: cloud not found device: {dev_path}")
