[文档]async def do(ws: WebSocket):
    """
    socket 处理器

    Args:
        ws(WebSocket):  websocket 对象
    """
    await ws.accept()
    while True:
        body = await ws.receive()
        ws_type = body["type"]

        if ws_type == "websocket.disconnect":
            print('websocket is disconnected...')
            break

        elif ws_type == "websocket.receive":
            message = json.loads(body["text"])
            if not websocket_validate.validate(message):
                await ws.send_text(f"error message is:{websocket_validate.errors}")
                continue

            await handle(ws, message)
