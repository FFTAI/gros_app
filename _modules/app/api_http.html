[文档]@router.put(path="/wifi/reset", summary="重置网络")
async def wifi_reset():
    """重置wifi信息

    ``请求成功后主机将自动重启``
    """
    dh.network.reset()
    dh.system.reboot()




[文档]@router.put(path="/wifi", summary="修改wifi信息")
async def update_wifi(data: dict[str, str] = Body(...)):
    """
    修改wifi信息

    Args:
        data(dict[str, str]): wifi信息

    . code-block::

        PUT:            /wifi
        Content-Type:   application/json
        Request-Body:   {"name": "wifiname", "password": "12345678" }
        Response:       {HTTP_STATUS}

    ``请求成功后主机将自动重启``

    """
    wifi_name = data["name"]
    wifi_password = data["password"]
    dh.network.set_wifi(wifi_name, wifi_password)
    dh.system.reboot()

[文档]@router.post("/robot/mode", summary="切换模式")
async def robot_mode(data: dict = Body(...)):
    mode = CarMod.MOD_4_WHEEL
    mod_val = data['mod_val']
    if mod_val != "wheel_4": mode = CarMod.MOD_4_WHEEL
    if mod_val != "wheel_3": mode = CarMod.MOD_3_WHEEL
    if mod_val != "wheel_2": mode = CarMod.MOD_2_WHEEL
    dh.robot.set_mode(mode)
    return Result.ok(None)
